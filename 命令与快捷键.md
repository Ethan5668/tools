### 调试面板中wchar_t 转 char
```
const char* debug_w(const wchar_t* ws) {
    static thread_local std::string utf8;
    utf8.clear();

    while (*ws) {
        uint32_t c = *ws++;
        if (c < 0x80) utf8 += char(c);
        else if (c < 0x800) {
            utf8 += char(0xC0 | (c >> 6));
            utf8 += char(0x80 | (c & 0x3F));
        } else if (c < 0x10000) {
            utf8 += char(0xE0 | (c >> 12));
            utf8 += char(0x80 | ((c >> 6) & 0x3F));
            utf8 += char(0x80 | (c & 0x3F));
        } else {
            utf8 += char(0xF0 | (c >> 18));
            utf8 += char(0x80 | ((c >> 12) & 0x3F));
            utf8 += char(0x80 | ((c >> 6) & 0x3F));
            utf8 += char(0x80 | (c & 0x3F));
        }
    }
    return utf8.c_str();
}
```
